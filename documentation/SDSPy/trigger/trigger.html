 98        """
 99        PySDS [Trigger][SetCoupling] :  Configure the source and coupling of the trigger
100

102

108                self.GetAllErrors() : List of errors

325    def SetPattern(self, Sources: list, Status: list, Pattern):
326        """
327        PySDS [Trigger][SetPattern] :  Configure a triggering pattern (Enable multi channel triggering)
328
329            Arguments :
330                Source : List of the sources used for the operation. Can only be C1 | C2 | C3 | C4
331                Status : List of the status for each source : X | L | H (X = don't care)
332                Pattern : AND | OR | NAND | NOR
333
334            Returns :
335                self.GetAllErrors() : List of errors
336        """
337        for Source in Sources:
338            if Source not in ["C1", "C2", "C3", "C4"]:
339                return [1, -1]  # Emulate the standard return type
340
341        for State in Status:
342            if State not in ["X", "L", "H"]:
343                return [1, -2]  # Emulate the standard return type
344
345        if Pattern not in ["AND", "OR", "NAND", "NOR"]:
346            return [1, -3]  # Emulate the standard return type
347
348        if len(Sources) != len(Status):
349            return [1, -3]  # Emulate the standard return type
350
351        if len(Sources) < 1:
352            return [1, -3]  # Emulate the standard return type
353
354        cmd = ""
355        for index in range(len(Sources)):
356            cmd += Sources[index] + "," + Status[index] + ","
357
358        self.__instr__.write(f"TRPA {cmd}STATE,{Pattern}")
359        return self.__baseclass__.GetAllErrors()
360
361    def GetPattern(self):
362        """
363        PySDS [Trigger][GetPattern] : Read the used pattern trigger
364
365            Arguments :
366                None
367
368            Returns :
369                List of Channel, Conditions and Pattern
370        """
371
372        Ret = self.__instr__.query("TRPA?").strip().split(" ")[-1].split(",")
373
374        Pattern = Ret[-1]
375        Sources = []
376        Conditions = []
377
378        for index in range(0, 2 * int((len(Ret) - 2) / 2), 2):
379            if Ret[index + 1] != "X":
380                Sources.append(Ret[index])
381                Conditions.append(Ret[index + 1])
382
383        return Sources, Conditions, Pattern
