 41    def GetSignalFrequency(self):
 42        """
 43        pySDS [Measure][GetSignalFrequency] : Get the number of trigger crossing per seconds, in Hz.
 44
 45        WARNING : Measure only the trigger channel.
 46
 47            Arguments :
 48                None
 49
 50            Returns :
 51                Measure, in Hz.
 52        """
 53        ret = self.__instr__.query("CYMT?").strip().split(" ")[-1]
 54
 55        if "E" in ret:
 56            # Notation 1 (seems newer ?)
 57            return float(ret[:-2])
 58        else:
 59            if "<" in ret:
 60                # Some old notation are <10Hz, with is casted to 10 and returned like that.
 61                # They removed it in the newest devices, with then use the parsing.
 62                return 10.00
 63
 64            val = ret.replace("k", "").replace("M", "")
 65            val = float(val[:-2].strip())
 66
 67            if "k" in ret:
 68                return val * 1_000
 69            if "M" in ret:
 70                return val * 1_000_000
 71            return val

429    def GetStatsMeasure(self, MeasureID):
430        """
431        pySDS [Measure][GetStatsMeasure] : Read the statistics for a measure.
432
433        WARNING : This function need statistics to be enabled !
434        WARNING2 : The ID of the measure
435
436            Arguments :
437                NumberOfMeasures : The number of measured stats wanted.
438
439            Returns :
440                dict with measure name and value.
441                Errors are reaturned under the error key.
442
443            Errors codes :
444                -1 : Invalid Number of measures. Must be between 1 and 5
445
446        """
447        if MeasureID < 0 or MeasureID > 5:
448            return {"errors": -1}
449
450        out = dict()
451        ret = (
452            self.__instr__.query(f"PAVA? STAT{MeasureID}").strip().split(" ")[2:]
453        )  # Two last elements
454        out["channel"] = ret[0]
455
456        measures = ret[1].split(":")
457        out["type"] = measures[0]
458
459        values = measures[1].split(",")
460
461        for index, meas in enumerate(values):
462            # We need to handle naming and value for each
463            if index % 2 == 0:
464                key = meas
465                val = values[index + 1]
466
467                # remove traling units
468                while val[-1].isalpha():
469                    val = val[:-1]
470
471                if "*" in val:
472                    val = 0.00
473
474                val = float(val)
475                out[key] = val
476            else:
477                # Continue the loop. We're not on the correct index
478                continue
479        return out

430    def GetStatsMeasure(self, MeasureID):
431        """
432        pySDS [Measure][GetStatsMeasure] : Read the statistics for a measure.
433
434        WARNING : This function need statistics to be enabled !
435        WARNING2 : The ID of the measure
436
437            Arguments :
438                NumberOfMeasures : The number of measured stats wanted.
439
440            Returns :
441                dict with measure name and value.
442                Errors are reaturned under the error key.
443
444            Errors codes :
445                -1 : Invalid Number of measures. Must be between 1 and 5
446
447        """
448        if MeasureID < 0 or MeasureID > 5:
449            return {"errors": -1}
450
451        out = dict()
452        ret = (
453            self.__instr__.query(f"PAVA? STAT{MeasureID}").strip().split(" ")[2:]
454        )  # Two last elements
455        out["channel"] = ret[0]
456
457        measures = ret[1].split(":")
458        out["type"] = measures[0]
459
460        values = measures[1].split(",")
461
462        for index, meas in enumerate(values):
463            # We need to handle naming and value for each
464            if index % 2 == 0:
465                key = meas
466                val = values[index + 1]
467
468                # remove traling units
469                while val[-1].isalpha():
470                    val = val[:-1]
471
472                if "*" in val:
473                    val = 0.00
474
475                val = float(val)
476                out[key] = val
477            else:
478                # Continue the loop. We're not on the correct index
479                continue
480        return out
