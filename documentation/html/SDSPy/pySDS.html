 79    def __init__(self, IP: str):
 80        """
 81        PySDS [init] :  Initialize the class.
 82                        Use some configuration file to initialize properly the oscilloscope, and read it's actual state to make sure to fetch the real state
 83                        May take some time since a lot of network requests are done here !
 84
 85            Arguments :
 86                IP : A string IP address, version 4 of where the ressource shall be allocated
 87
 88            Returns :
 89                None
 90        """
 91
 92        # First, validate the IP and try to open the ressource !
 93        try:
 94            self.__ip__ = ipaddress.ip_address(IP)
 95        except ValueError:
 96            print(
 97                "     [ PySDS ] [ Init ] : Incorrect IP was passed to the constructor"
 98            )
 99            self.DeviceOpenned = 0
100            return
101
102        try:
103            self.__rm__ = pyvisa.ResourceManager()
104            self.__instr__ = self.__rm__.open_resource(f"TCPIP0::{IP}::inst0::INSTR")
105        except:
106            print(
107                "     [ PySDS ] [ Init ] : Unable to access to the device. Check if the IP is right, or if you can ping it !"
108            )
109            self.DeviceOpenned = 0
110            return
111
112        # Then, request for the IDN command.
113        # Typical return : Siglent Technologies,SDS824X HD,SDS08A0C802019,3.8.12.1.1.3.8
114        IDN = self.__instr__.query("*IDN?")
115        IDN = IDN.split(",")
116
117        # Check if the brand is the right one, or this library isn't going to work !
118        if IDN[0].find("Siglent") == -1:
119            print("     [ PySDS ] [ Init ] : Found a non Siglent Device on this IP !")
120            self.DeviceOpenned = 0
121            return
122
123        # Parse some different fields
124        self.model = IDN[1]
125        self.SN = IDN[2]
126        self.Firmware = IDN[3]
127
128        # Load the right configuration file
129        # First, replace any space in the name with a "-" to ensure compatibility within different OS
130        self.model = self.model.replace(" ", "-")
131
132        # Load the right configuration file without the SDS in front
133        self.__ConfigFile__ = self.model[3:] + ".toml"
134
135        self.__Config__ = None
136        with open(f"config/{self.__ConfigFile__}", "rb") as f:
137            self.__Config__ = tomllib.load(f)
138
139        # Create a generic class, for internal usage only
140        self.__Generics__ = SCPIGenerics(self.__instr__, self)
141
142        # Now, initialize some parameters from the configuration file
143        self.Channel = []
144        for index in range(self.__Config__["Specs"]["Channel"]):
145            self.Channel.append(
146                SiglentChannel(
147                    self.__instr__, self, f"C{index}", self.__Config__["impedance"]
148                )
149            )
150
151        # Then, initialize all of the subclass
152        
153        self.Acquistion = SiglentAcquisition(self.__instr__, self)
154        """SiglentAcquisition subclass. Check dedicated documentation :`SDSPy.acquisition.SiglentAcquisition`"""
155
156        self.Autotest = SiglentPassFail(self.__instr__, self)
157        """SiglentPassFail subclass. Check dedicated documentation"""
158
159        self.Communication = SiglentCommunication(self.__instr__, self)
160        """SiglentCommunication subclass. Check dedicated documentation"""
161
162        self.Cursor = SiglentCursor(self.__instr__, self)
163        """SiglentCursor subclass. Check dedicated documentation"""
164
165        self.Decode = SiglentDecode(self.__instr__, self)
166        """SiglentDecode subclass. Check dedicated documentation"""
167
168        self.Digital = SiglentDigital(self.__instr__, self)
169        """SiglentDigital subclass. Check dedicated documentation"""
170
171        self.Display = SiglentScreen(self.__instr__, self)
172        """SiglentScreen subclass. Check dedicated documentation"""
173
174        self.Files = SiglentFiles(self.__instr__, self)
175        """SiglentFiles subclass. Check dedicated documentation"""
176
177        self.History = SiglentHistory(self.__instr__, self)
178        """SiglentHistory subclass. Check dedicated documentation"""
179
180        self.Maths = SiglentMaths(self.__instr__, self)
181        """SiglentMaths subclass. Check dedicated documentation"""
182
183        self.Measure = SiglentMeasure(self.__instr__, self)
184        """SiglentMeasure subclass. Check dedicated documentation"""
185
186        self.Reference = SiglentReference(self.__instr__, self)
187        """SiglentReference subclass. Check dedicated documentation"""
188
189        self.Trigger = SiglentTrigger(self.__instr__, self)
190        """SiglentTrigger subclass. Check dedicated documentation"""
191
192        self.Timebase = SiglentTimebase(self.__instr__, self)
193        """SiglentTimebase subclass. Check dedicated documentation"""
194
195        self.Waveform = SiglentWaveform(self.__instr__, self)
196        """SiglentWaveform subclass. Check dedicated documentation"""
197
198
199        # For some older device, load additionnal commands that are depecrated in the newest models / firmwares
200        if "ACAL" in self.__Config__["Specs"]["LegacyFunctions"]:
201            self.Calibration = ACAL(self.__instr__, self)
202
203        if "DATE" in self.__Config__["Specs"]["LegacyFunctions"]:
204            self.Date = DATE(self.__instr__, self)
205
206        if "COUNTER" in self.__Config__["Specs"]["LegacyFunctions"]:
207            self.Counter = COUNTER(self.__instr__, self)
208
209        # Then, load default settings by sending request to get the actual state of the device
210
211        # Check for support of depecrated commands !
212        # ACAL (SDS2000X and SDS1000CFL)
213        # AAUTS (Up to SDS1000X)
214        # COUNTER (SDS1000 non SPO)
215        # CSV SAVE (All but different formats...)
216        # DATE (SDS2000X and SDS1000CFL)
217        # FFTZOOM (Up to SDS1000X)
218        # FILTER (SDS1000 non SPO)
219        # FILT SET (SDS1000 non SPO)
220        # PEAK DETECT (Up to SDS1000X)
221        # PFCT (Up to SDS1000X)
222        # PERS (Up to SDS1000X)
223        # RECALL (Up to SDS1000X)
224        # REFSET (Up to SDS1000X)
225        # VPOS (Up to SDS1000X)
226
227        self.DeviceOpenned = 1
228        return

501    def GetAllErrors(self, print=False):
502        """
503        PySDS [GetAllErrors] :  Read the device errors, and until at least one error exist, continue to read it.
504                                For each errors, it will be printed in console and returned on a list, with it's lengh in first position.
505
506                                This function also trigger a reading of the status of the device to detect if value where adapted or cancelled.
507
508            Arguments :
509                print : Shall we print the decoded output on the console ? Default to false.
510
511            Returns :
512                List :
513                    Index 0 :       Number of errors that occured
514                    Index 1 - n :   Device errors codes
515        """
516
517        FetchNextError = True
518        Errors = [0]
519
520        # For each loop, we ask the device an error
521        # If not 0, then we parse it and add it to the list
522        # When the last error has been fetched (or no errors at all !), we exit the loop
523
524        while FetchNextError:
525            Ret = self.__Generics__.ReadEXR()
526
527            if Ret == 0:
528                FetchNextError = False
529
530            else:
531                Errors[0] += 1
532                Errors.append(int(Ret))
533
534                if print == True:
535                    # Theses errors messages came from the Siglent SCPI documentation, and are only here to help the developper to get the error easily !
536                    match Ret:
537                        case 21:
538                            print(
539                                f"     [ PySDS ] [ GetAllErrors ] : ({Ret}) Permission error. The command cannot be executed in local mode."
540                            )
541                        case 22:
542                            print(
543                                f"     [ PySDS ] [ GetAllErrors ] : ({Ret}) Environment error. The instrument is not configured to correctly process command. For instance, the oscilloscope cannot be set to RIS at a slow timebase."
544                            )
545                        case 23:
546                            print(
547                                f"     [ PySDS ] [ GetAllErrors ] : ({Ret}) Option error. The command applies to an option which has not been installed."
548                            )
549                        case 25:
550                            print(
551                                f"     [ PySDS ] [ GetAllErrors ] : ({Ret}) Parameter error. Too many parameters specified."
552                            )
553                        case 26:
554                            print(
555                                f"     [ PySDS ] [ GetAllErrors ] : ({Ret}) Non-implemented command."
556                            )
557                        case 32:
558                            print(
559                                f"     [ PySDS ] [ GetAllErrors ] : ({Ret}) Waveform descriptor error. An invalid waveform descriptor has been detected."
560                            )
561                        case 36:
562                            print(
563                                f"     [ PySDS ] [ GetAllErrors ] : ({Ret}) Panel setup error. An invalid panel setup data block has been detected."
564                            )
565                        case 50:
566                            print(
567                                f"     [ PySDS ] [ GetAllErrors ] : ({Ret}) No mass storage present when user attempted to access it."
568                            )
569                        case 53:
570                            print(
571                                f"     [ PySDS ] [ GetAllErrors ] : ({Ret}) Mass storage was write protected when user attempted to create, or a file, to delete a file, or to format the device."
572                            )
573                        case 58:
574                            print(
575                                f"     [ PySDS ] [ GetAllErrors ] : ({Ret}) Mass storage file not found."
576                            )
577                        case 59:
578                            print(
579                                f"     [ PySDS ] [ GetAllErrors ] : ({Ret}) Requested directory not found."
580                            )
581                        case 61:
582                            print(

584                            )
585                        case 62:
586                            print(
587                                f"     [ PySDS ] [ GetAllErrors ] : ({Ret}) Cannot write on mass storage because filename already exists."
588                            )
589
590        # When the loop exist, we return the list
591        Retval = self.GetDeviceStatus(print)
592
593        if Retval != 0:
594            Errors[0] += 1
595            Errors.append(
596                Retval + 1000
597            )  # Increment of 1000 to signify an error in the MSB register
598
599        return Errors

601    def GetDeviceStatus(self, print=False):
602        """
603        PySDS [GetDeviceStatus] :   Get the device status, and parse it to make it easier to use for developpers or users.
604                                    Print each status bit
605
606            Argument :
607                print : Shall we print the decoded output on the console ? Default to false.
608
609            Returns :
610                List of lenght 16, for each bit
611        """
612
613        # Fetch the value
614        Ret = self.__Generics__.ReadINR()
615
616        # Mask each bit in the range.
617        # We do this by logic AND and shifting to get back to 0 | 1
618        Bits = []
619        for power in range(16):
620            Bits.append((Ret & pow(2, power)) >> power)
621
622        print("Device status :")
623        print("Bit | Status | Message")
624        for index, bit in enumerate(Bits):
625            match index:
626                case 0:
627                    message = "A new signal has been acquired"
628                case 1:
629                    message = "A screen dump has terminated"
630                case 2:
631                    message = "A return to the local state is detected"
632                case 3:
633                    message = "A time-out has occurred in a data block transfer"
634                case 4:
635                    message = "A segment of a sequence waveform has been acquired"
636                case 5:
637                    message = "Reserved for LeCroy use"
638                case 6:
639                    message = 'Memory card, floppy or hard disk has become full in "AutoStore Fill" mode'
640                case 7:
641                    message = (
642                        "A memory card, floppy or hard disk exchange has been detected"
643                    )
644                case 8:
645                    message = "Waveform processing has terminated in Trace A"
646                case 9:
647                    message = "Waveform processing has terminated in Trace B"
648                case 10:
649                    message = "Waveform processing has terminated in Trace C"
650                case 11:
651                    message = "Waveform processing has terminated in Trace D"
652                case 12:
653                    message = "Pass/Fail test detected desired outcome"
654                case 13:
