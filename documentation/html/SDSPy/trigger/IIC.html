 39        """
 40        pySDS [IIC][SetTriggerOnSCL] : Configure the trigger on the SCL bus.
 41
 42            Arguments :
 43                Channel :       SiglentChannel or SiglentDCHannel related to the SCL pin
 44                Threshold :     For analog channel only, the used voltage. Default to 1.65
 45
 46            Returns :
 47                self.GetAllErrors()
 48                or
 49                -1 : Invalid Channel
 50        """
 51        if type(Channel) is not SiglentChannel and type(Channel) is not SiglentDChannel:

 53
 54        cmd = f"TRIIC:SCL {Channel.__channel__}"
 55
 56        if type(Channel) == SiglentChannel:
 57            cmd += f",{Threshold}"
 58
 59        self.__instr__.write(cmd)

 87        """
 88        pySDS [IIC][SetTriggerOnCondition] : Set the trigger on a bus transfer condition
 89
 90        WARNING : Once the condition is selected, further configuration may be needed.
 91
 92            Arguments :
 93                Condition : The selected condition
 94
 95            Returns :
 96                self.GetAllErrors()
 97                or
 98                -1 : Invalid condition
 99
100            Possibles conditions :
101                START : Start condition.
102                STOP : Stop condition.
103                RESTART : Another start condition occurs before a stop condition.
104                NOACK : Missing acknowledge.
105                EEPROM : EEPROM frame containing (Start:Controlbyte:R:Ack:Data).
106                7ADDA : 7-bit address frame containing (Start:Address7:R/W:Ack:Data:Data2).
107                10ADDA : 10-bit address frame containing (Start:Address10:R/W:Ack:Data:Data2).
108                DALENTH : specifie a search based on address lengthand data length.
109        """
110        if Condition not in [
111            "START",
112            "STOP",
113            "RESTART",
114            "NOACK",
115            "EEPROM",
116            "7ADDA",
117            "10ADDA",
118            "DALENTH",
119        ]:
120            return [1, -1]
121
122        self.__instr__.write(f"TRIIC:CON {Condition}")
