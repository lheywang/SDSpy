 36    def SetTriggerOnCANH(self, Channel, Threshold=1.65):
 37        """
 38        pySDS [CAN][SetTriggerOnCANH] : Configure the trigger on the CANH pin
 39
 40            Arguments :
 41                Channel :       SiglentChannel or SiglentDCHannel related to the CANH pin
 42                Threshold :     For analog channel only, the used voltage. Default to 1.65
 43
 44            Returns :
 45                self.GetAllErrors()
 46                or
 47                -1 : Invalid Channel
 48        """
 49        if type(Channel) is not SiglentChannel and type(Channel) is not SiglentDChannel:
 50            return [1, -1]
 51
 52        cmd = f"TRCAN:CANH {Channel.__channel__}"
 53
 54        if type(Channel) == SiglentChannel:
 55            cmd += f",{Threshold}"
 56
 57        self.__instr__.write(cmd)
 58        return self.__baseclass__.GetAllErrors()

 60    def SetTriggerOnCondition(self, Condition: str):
 61        """
 62        pySDS [CAN][SetTriggerOnCondition] : Configure the trigger on a condition on the CAN Bus
 63
 64            Arguments :
 65                Condition : START | REMOTE | ID | ID_AND_DATA | ERROR
 66
 67            Returns :
 68                self.GetAllErrors()
 69                or
 70                -1 : Invalid condition
 71
 72            Possibles conditions
 73                START— Start condition.
 74                REMOTE— Remote frame
 75                ID— Specifies a search based on ID bits and ID.
 76                ID_AND_DATA— Specify a search based on ID bits, ID and data.
 77                ERROR— Error frame
 78        """
 79        if Condition not in ["START", "REMOTE", "ID", "ID_AND_DATA", "ERROR"]:
 80            return [1, -1]
 81
 82        self.__instr__.write(f"TRCAN:CON {Condition}")
 83        return self.__baseclass__.GetAllErrors()

153        """
154        pySDS [CAN][SetTriggerBaud] : Set the baud rate of the UART comm
155
156            Arguments :
157                Baud : Value. (As a string for standard values, or as an integer for non standard)*
158
159                * Sending a standard as integer will be seen as custom but will also works fine !
160
161            Returns :
162                self.GetAllErrors()
163                or
164                -1 : Invalid Baud
165
166            Standard baud values :
167                5k
168                10k

170                50k
171                100k
172                125k
173                250k
174                500k
175                800k
176                1M
177        """
178        if Baud < 5_000 or Baud > 1_000_000:
179            return [1, -1]
180
181        if Baud not in [
182            "5k",
183            "10k",
184            "20k",
185            "59k",
186            "100k",
187            "125k",
188            "250k",
189            "500k",
190            "800k",
191            "1M",
192        ]:
193            cmd = f"TRCAN:BAUD CUSTOM,{Baud}"
194        else:
195            cmd = f"TRCAN:BAUD {Baud}"
196
197        self.__instr__.write(cmd)
198        return self.__baseclass__.GetAllErrors()
